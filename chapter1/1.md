void Descend(int &a, int &b, int &c)  // 通过交换，令 a >= b >= c
{   // Add your code here
    int temp;
    if (a < b) {
        temp = a;
        a = b;
        b = temp;
    }
    if (b < c) {
        temp = b;
        b = c;
        c = temp;
    }

        b = temp;

}

##   DC01PE11 求k阶裴波那契序列的第m项的值
```c
Status Fibonacci(int k, int m, int &f) { 
    // Add your code here
    if (k <= 0 || m < 0) {  
        return ERROR;
    }

    // 如果 k == 1，且 m == 0，f = 0；否则返回 ERROR
    if (k == 1) {
        if (m == 0) {
            f = 0;
            return OK;
        } else {
            return ERROR;
        }
    }

    // 如果 m <= k-2，那么根据题目定义 f=0
    if (m <= k - 2) {
        f = 0;
        return OK;
    }

    // 如果 m == k-1，f=1
    if (m == k - 1) {
        f = 1;
        return OK;
    }
    
    int *fib=new int[m+1];
    for (int i = 0; i < k-1; i++) {
        fib[i]=0;
    }
    fib[k-1]=1;
    for (int i = k; i <= m; i++) {
        fib[i]=0;
        for (int j = 1; j <= k; j++) {
            fib[i]+=fib[i-j];
        }
    }
    f=fib[m];
    delete[] fib;

    return OK;
}
```

        return NULL;
    }
